async function exportVideo() { if(isPlaying) togglePlay(); saveFrame(); UI.status.classList.remove('hidden'); UI.status.classList.add('flex'); UI.statusText.innerText = "Подготовка кадров..."; try { const fps = parseInt(UI.fps.value) || 12; const tempC = document.createElement('canvas'); tempC.width = CANVAS_WIDTH; tempC.height = CANVAS_HEIGHT; const tCtx = tempC.getContext('2d'); let totalFrames = frames.length; audioClips.forEach(c => { const end = c.startTimeFrames + (c.durationSec * fps); if(end > totalFrames) totalFrames = Math.ceil(end); }); // Кэшируем кадры, чтобы не было просадок FPS и рассинхрона со звуком const loadedImages = []; const preloadPromises = frames.map((frameDataUrl, index) => { return new Promise((resolve) => { const img = new Image(); img.onload = () => { loadedImages[index] = img; resolve(); }; img.src = frameDataUrl; }); }); await Promise.all(preloadPromises); for (let i = frames.length; i < totalFrames; i++) { loadedImages[i] = loadedImages[frames.length - 1]; } // Принудительно заставляем систему кодировать в MP4 контейнер let mime = 'video/mp4;codecs=avc1,mp4a.40.2'; if (!MediaRecorder.isTypeSupported(mime)) { mime = 'video/webm;codecs=vp8,opus'; // Фолбек, если MP4 совсем не поддерживается WebView } const stream = tempC.captureStream(fps); const destNode = audioCtx.createMediaStreamDestination(); // Подмешиваем аудиодорожку прямо в поток видео рекордера if (audioClips.length > 0) { destNode.stream.getAudioTracks().forEach(t => stream.addTrack(t)); } const recorder = new MediaRecorder(stream, { mimeType: mime, videoBitsPerSecond: 2500000, audioBitsPerSecond: 128000 }); const chunks = []; recorder.ondataavailable = e => { if(e.data.size) chunks.push(e.data); }; recorder.onstop = async () => { const blob = new Blob(chunks, {type: mime}); const filename = `ZRDraw_${Date.now()}.mp4`; if (window.Android && window.Android.saveFile) { UI.statusText.innerText = "Передача в Галерею..."; const reader = new FileReader(); reader.onloadend = function() { // Передаем полную строку обратно в Sketchware Java window.Android.saveFile(reader.result, filename); UI.status.classList.add('hidden'); UI.status.classList.remove('flex'); loadFrameToCanvas(0); }; reader.readAsDataURL(blob); } else { const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = filename; document.body.appendChild(a); a.click(); document.body.removeChild(a); UI.status.classList.add('hidden'); UI.status.classList.remove('flex'); loadFrameToCanvas(0); } }; // Запускаем запись. Шаг в 100мс принудительно заставляет писать индексы таймлайна кадра recorder.start(100); if (audioCtx.state === 'suspended') await audioCtx.resume(); const now = audioCtx.currentTime; // Включаем звук для записи в поток audioClips.forEach(clip => { const source = audioCtx.createBufferSource(); source.buffer = clip.buffer; source.connect(destNode); source.start(now + (clip.startTimeFrames / fps), clip.trimStartSec, clip.durationSec); }); let exportFrame = 0; const frameDuration = 1000 / fps; function renderNextFrame() { if (exportFrame >= totalFrames) { // Важно: ждем 400мс после окончания мультика, чтобы буфер аудио успел полностью дойти до файла setTimeout(() => { recorder.stop(); }, 400); return; } UI.statusText.innerText = `Рендер кадра: ${exportFrame+1}/${totalFrames}`; tCtx.fillStyle = bgColor; tCtx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT); if (loadedImages[exportFrame]) { tCtx.drawImage(loadedImages[exportFrame], 0, 0, CANVAS_WIDTH, CANVAS_HEIGHT); } exportFrame++; setTimeout(renderNextFrame, frameDuration); } renderNextFrame(); } catch(e) { UI.status.classList.add('hidden'); UI.status.classList.remove('flex'); alert("Ошибка: " + e.message); } }
